59double score_english_hex(const char *hex) {
60 size_t hex_len = strlen(hex);
61 if (hex_len == 0 || (hex_len & 1)) {
62 // empty or odd-length hex is invalid
63 return -1e9;
64 }
65
66 int counts[27] = {0}; // letter+space counts
67 size_t total_letters = 0;
68 size_t total_bytes = 0;
69 double penalty = 0.0;
70
71 // Decode hex on the fly; no need to allocate a separate buffer.
72 for (size_t i = 0; i < hex_len; i += 2) {
73 int hi = hex_value(hex[i]);
74 int lo = hex_value(hex[i + 1]);
75 if (hi < 0 || lo < 0) {
76 // invalid hex character
77 return -1e9;
78 }
79
80 unsigned char c = (unsigned char)((hi << 4) | lo);
81 total_bytes++;
82
83 if (isalpha(c)) {
84 c = (unsigned char)tolower(c);
85 counts[c - 'a']++;
86 total_letters++;
87 } else if (c == ' ') {
88 counts[26]++;
89 total_letters++;
90 } else if (c == '\n' || c == '\r' || c == '\t' ||
91 c == ',' || c == '.' || c == '\'' || c == '"') {
92 // Neutral punctuation/whitespace: allowed but not counted as letters.
93 } else if (c < 32 || c > 126) {
94 // Non-printable or non-ASCII characters: penalize heavily but keep scoring.
95 penalty += 50.0;
96 continue;
97 } else {
98 // Other printable symbols like ! ? ; : etc. are allowed but not counted as letters.
99 }
100 }
101
102 if (total_bytes == 0) {
103 return -1e9;
104 }
105 if (total_letters == 0) {
106 return -1000.0 - penalty;
107 }
108
109 // Compute a chi-squared style statistic over letters+space.
110 // Lower chi2 means closer to English; we will invert it into a score.
111 double chi2 = 0.0;
112 for (int i = 0; i < 27; i++) {
113 double expected = english_freq[i] * (double)total_letters;
114 double observed = (double)counts[i];
115 double diff = observed - expected;
116 // Add a tiny constant to avoid division by zero.
117 chi2 += (diff * diff) / (expected + 1e-9);
118 }
119
120 // Convert chi-squared to a score.
121 // Smaller chi2 -> higher score. Add bonus for high proportion of letters/spaces.
122 double letter_ratio = (double)total_letters / (double)total_bytes;
123 double score = -chi2 + letter_ratio * 50.0 - penalty;
124
125 return score;
126}
