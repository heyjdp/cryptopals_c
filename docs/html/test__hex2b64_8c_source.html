11/*
12 * Run hex2b64_stream() over an in-memory string and capture its output in out.
13 */
14static int convert_hex_string(const char *hex, char *out, size_t out_cap) {
15 FILE *in = tmpfile();
16 FILE *outf = tmpfile();
17 if (!in || !outf) {
18 if (in) fclose(in);
19 if (outf) fclose(outf);
20 return -1;
21 }
22
23 fputs(hex, in);
24 rewind(in);
25
26 int rc = hex2b64_stream(in, outf);
27
28 fflush(outf);
29 rewind(outf);
30
31 size_t n = fread(out, 1, out_cap - 1, outf);
32 out[n] = '\0';
33
34 fclose(in);
35 fclose(outf);
36
37 return rc;
38}
